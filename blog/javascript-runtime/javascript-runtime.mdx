## Starting Problem

```js
console.log('Start');

const foo = () => {
    Promise.resolve('Promise')
        .then((msg) => console.log(msg));
};

const bar = () => {
    setTimeout(() => console.log('Timer'), 0);
    return foo();
};

bar();
console.log('End');

```

## Stack Explanation

```js
console.log('Start');

const foo = () => {
    console.log('foo');
};

const bar = () => {
    console.log('bar');
    return foo();
};

bar();
console.log('End');
```

### Final Problem

```js
console.log('Start');

const foo = () => {
    Promise.resolve('Microtask in Queue')
        .then((msg) => console.log(msg));
};

const bar = () => {
    setTimeout(() => console.log('Timeout in bar'), 100);
    return foo();
};

bar();
console.log('End');
```
